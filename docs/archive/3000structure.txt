	const [mouthOpen, setMouthOpen] = useState(0);

	const [lipData, setLipData] = useState({ volume: 0 });

		

	useEffect(() =&gt; {

	navigator.mediaDevices

	.getUserMedia({ audio: true })

	.then((stream) =&gt; {

	const ctx = new AudioContext();

	const analyser = ctx.createAnalyser();

	const source = ctx.createMediaStreamSource(stream);

	source.connect(analyser);

	analyser.fftSize = 1024;

		

	const dataArray = new Uint8Array(analyser.frequencyBinCount);

		

	const loop = () =&gt; {

	analyser.getByteTimeDomainData(dataArray);

	let sum = 0;

	for (let i = 0; i &lt; dataArray.length; i++) {

	sum += Math.abs(dataArray[i] - 128);

	}

	const volume = sum / dataArray.length / 20;

		

	setMouthOpen(volume);

	setLipData({ volume });

		

	requestAnimationFrame(loop);

	};

		

	loop();

	});

		

	return () =&gt; {};

	}, []);

		

	return { mouthOpen, lipData };

	}

	let broadcaster: WebSocket | null = null;

	const viewers = new Set&lt;WebSocket&gt;();

		

	export const GET = async (req: NextRequest) =&gt; {

	const { socket } = (req as any).webSocket;

	socket.accept();

		

	socket.onmessage = (msg: any) =&gt; {

	const data = JSON.parse(msg.data);

		

	if (data.role === &quot;broadcaster&quot;) {

	broadcaster = socket;

	return;

	}

		

	if (data.role === &quot;viewer&quot;) {

	viewers.add(socket);

	if (broadcaster) {

	broadcaster.send(JSON.stringify({ type: &quot;viewer-joined&quot; }));

	}

	return;

	}

		

	if (data.type === &quot;offer&quot; &amp;&amp; broadcaster) {

	broadcaster.send(JSON.stringify({ type: &quot;offer&quot;, offer: data.offer }));

	}

		

	if (data.type === &quot;answer&quot;) {

	for (const v of viewers) {

	v.send(JSON.stringify({ type: &quot;answer&quot;, answer: data.answer }));

	}

	}

		

	if (data.type === &quot;candidate&quot;) {

	if (broadcaster) broadcaster.send(JSON.stringify({ type: &quot;candidate&quot;, candidate: data.candidate }));

	for (const v of viewers) v.send(JSON.stringify({ type: &quot;candidate&quot;, candidate: data.candidate }));

	}

	};

		

	socket.onclose = () =&gt; {

	viewers.delete(socket);

	if (socket === broadcaster) broadcaster = null;

	};

		

	return new Response(null, { status: 101 });

	};

	export function applyPhysics(avatar, senses) {

	const dx = cursor.x - avatar.x;
